```typescript
// ─── Phase 2: ArticleSnapshot ────────────────────────────────────────────────

/**
 * Zod schema for the article snapshot returned by the capture extractor.
 * All 5 fields are required; empty strings are valid (EXTRACTION_EMPTY handles
 * the case where both title and content are empty).
 *
 * Field ownership (CAP-03, CAP-04):
 *   title, description, content — provided by extractor (content script)
 *   url, create_at             — provided by SW from tabs.query result
 */
export const ArticleSnapshotSchema = z.object({
  title:       z.string(),               // Readability.parse().title || document.title
  url:         z.string().url(),         // SW: tab.url from tabs.query
  description: z.string(),              // meta[name=description] → og:description → Readability.excerpt
  create_at:   z.string().datetime(),   // SW: new Date().toISOString() at click time (CAP-04)
  content:     z.string(),              // DOMPurify(rawHtml) → Turndown Markdown (D-18)
});

export type ArticleSnapshot = z.infer<typeof ArticleSnapshotSchema>;
```

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-02-02-01 | Tampering | ArticleSnapshot.url 字段 | mitigate | z.string().url() 严格校验——非法 URL 在 zod parse 阶段被拦截（SW pipeline 第 7 步），不会传到 popup |
