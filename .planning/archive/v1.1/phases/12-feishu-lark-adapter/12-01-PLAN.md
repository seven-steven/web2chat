```typescript
export interface Snapshot {
  title: string;
  url: string;
  description: string;
  create_at: string;
  content: string;
}

export function composeFeishuMessage(payload: {
  prompt: string;
  snapshot: Snapshot;
  timestampLabel: string;
}): string {
  const { prompt, snapshot, timestampLabel } = payload;
  const lines: string[] = [];
  if (prompt) lines.push(prompt, '');
  if (snapshot.title) lines.push(snapshot.title, '');
  if (snapshot.url) lines.push(snapshot.url, '');
  if (snapshot.description) lines.push(snapshot.description, '');
  if (snapshot.create_at) lines.push(`${timestampLabel} ${snapshot.create_at}`, '');
  if (snapshot.content) lines.push(snapshot.content);
  return lines.join('\n').trim();
}
```

Key differences from telegram-format.ts:
- NO `TELEGRAM_CHAR_LIMIT` constant
- NO `TRUNCATION_SUFFIX` constant
- NO truncation logic ŌĆö function is pure field assembly + join + trim
- No markdown stripping or escaping ŌĆö plain text passthrough

Run tests ŌĆö they MUST pass.
