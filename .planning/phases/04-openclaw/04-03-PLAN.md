2. **Create `entrypoints/options/components/GrantedOriginsSection.tsx`**:

   ```typescript
   import { signal, effect } from '@preact/signals';
   import { t } from '@/shared/i18n';
   import * as grantedOriginsRepo from '@/shared/storage/repos/grantedOrigins';

   const originsSig = signal<string[]>([]);

   // Load origins on first render
   effect(() => {
     void grantedOriginsRepo.list().then((origins) => {
       originsSig.value = origins;
     });
   });

   export function GrantedOriginsSection() {
     async function handleRemove(origin: string) {
       await chrome.permissions.remove({ origins: [origin + '/*'] });
       await grantedOriginsRepo.remove(origin);
       originsSig.value = await grantedOriginsRepo.list();
     }

     const origins = originsSig.value;

     return (
       <section
         class="bg-slate-100 dark:bg-slate-800 rounded-lg p-6 flex flex-col gap-4"
         data-testid="options-origins-section"
       >
         <header class="flex flex-col gap-2">
           <h2 class="m-0 text-base leading-snug font-semibold text-slate-900 dark:text-slate-100">
             {t('options_origins_heading')}
           </h2>
           <p class="m-0 text-sm leading-normal font-normal text-slate-500 dark:text-slate-400">
             {t('options_origins_explainer')}
           </p>
         </header>
         {origins.length === 0 ? (
           <p
             class="m-0 text-sm leading-normal font-normal text-slate-500 dark:text-slate-400 italic"
             data-testid="options-origins-empty"
           >
             {t('options_origins_empty')}
           </p>
         ) : (
           <ul class="list-none m-0 p-0 flex flex-col gap-2">
             {origins.map((origin) => (
               <li
                 key={origin}
                 class="flex items-center justify-between bg-white dark:bg-slate-700 px-4 py-2 rounded-md"
                 data-testid={`options-origin-item-${origin}`}
               >
                 <span class="text-sm font-mono text-slate-700 dark:text-slate-200 truncate">
                   {origin}
                 </span>
                 <button
                   type="button"
                   class="text-red-600 hover:text-red-700 dark:text-red-400 dark:hover:text-red-300 text-sm font-semibold ml-4 shrink-0"
                   onClick={() => void handleRemove(origin)}
                   data-testid={`options-origin-remove-${origin}`}
                 >
                   {t('options_origins_remove')}
                 </button>
               </li>
             ))}
           </ul>
         )}
       </section>
     );
   }
   ```

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Popup -> chrome.permissions.request | User gesture triggers permission dialog — user makes explicit grant/deny decision |
| Options page -> chrome.permissions.remove | User explicitly removes a previously granted origin |
| SW -> chrome.permissions.contains | Defensive check after popup already granted — validates permission still active |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-04-03-01 | Elevation of Privilege | SendForm.tsx chrome.permissions.request | mitigate | Permission requested only for the specific origin from user-provided URL (not wildcards beyond `origin/*`); requires active user gesture (click) per Chrome API contract. |
| T-04-03-02 | Tampering | GrantedOriginsSection.tsx handleRemove | mitigate | Remove calls both `chrome.permissions.remove` (revokes browser-level permission) AND `grantedOriginsRepo.remove` (cleans storage). Double-action prevents stale grants. |
| T-04-03-03 | Information Disclosure | ErrorBanner OPENCLAW_* codes | accept | Error messages only show origin string (user-provided URL host). No internal implementation details exposed. Per Phase 2 convention, raw error.message never rendered to DOM. |
| T-04-03-04 | Denial of Service | dispatch-pipeline permissions.contains | accept | Synchronous Chrome API call with no timeout risk. If Chrome permissions store is corrupt, fails fast with rejection (caught by wrapHandler top-level). |
</threat_model>
