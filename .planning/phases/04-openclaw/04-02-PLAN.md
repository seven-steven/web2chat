```typescript
  type: 'ADAPTER_DISPATCH';
  payload: {

  };
}

interface AdapterDispatchResponse {
  ok: boolean;
  code?: string;
  message?: string;
  retriable?: boolean;
}

function isAdapterDispatch(msg: unknown): msg is AdapterDispatchMessage { ... }

export default defineContentScript({
  matches: [],
  registration: 'runtime',
  main() {
    chrome.runtime.onMessage.addListener((msg, _sender, sendResponse) => {
      if (!isAdapterDispatch(msg)) return false;
      void handleDispatch(msg.payload).then(sendResponse);
      return true;
    });
  },
});
```

```typescript
/**
 * OpenClaw adapter content script (ADO-01..ADO-04, D-39..D-53).
 *
 * Injected by SW dispatch-pipeline via chrome.scripting.executeScript
 * into the OpenClaw WebChat tab. Registers one-shot ADAPTER_DISPATCH
 * message listener following the same protocol as mock-platform.
 *
 * DOM strategy (D-47): CSS selector for textarea.
 * Injection (D-48): setInputValue property-descriptor setter.
 * Send (D-49): Enter keydown.
 * Confirm (D-50): MutationObserver watches for new message node.
 * Timeout: 5s for waitForReady + 5s for send confirmation (D-51).
 */
import { defineContentScript } from '#imports';
import { setInputValue } from '@/shared/dom-injector';
```

Content script structure:
- Constants:
  - `OPENCLAW_FEATURE_SELECTOR = '[data-testid="openclaw-app"], .agent-chat__messages, [class*="agent-chat"]'` (fallback chain per RESEARCH Open Q1 — checks title pattern too)
  - `TEXTAREA_SELECTOR = '.agent-chat__input textarea, textarea[data-testid="chat-input"]'`
  - `MESSAGE_LIST_SELECTOR = '.agent-chat__messages, [data-testid="message-list"]'`
  - `WAIT_TIMEOUT_MS = 5000`

- `isAdapterDispatch(msg)` type guard (same pattern as mock-platform)

- `AdapterDispatchMessage` and `AdapterDispatchResponse` interfaces (same as mock-platform but with extended error codes: `'OPENCLAW_OFFLINE' | 'INPUT_NOT_FOUND' | 'TIMEOUT'`)

- `defineContentScript({ matches: [], registration: 'runtime', main() { ... } })`:
  - Register one-shot `chrome.runtime.onMessage.addListener`
  - Guard with `isAdapterDispatch`, call `handleDispatch(msg.payload).then(sendResponse)`, return true

- `async function handleDispatch(payload)`:
  1. **canDispatch check (D-52, D-53):** Check for OpenClaw feature DOM via `document.querySelector(OPENCLAW_FEATURE_SELECTOR)` OR `document.title.toLowerCase().includes('openclaw')`. If NEITHER found, return `{ ok: false, code: 'OPENCLAW_OFFLINE', message: 'OpenClaw UI not detected on this page', retriable: true }`
  2. **waitForReady (D-51):** Call `waitForElement(TEXTAREA_SELECTOR, WAIT_TIMEOUT_MS)`. If null, return `{ ok: false, code: 'INPUT_NOT_FOUND', message: 'Chat input textarea not found', retriable: true }`
  3. **compose (D-39, D-40, D-41):** Call `composeMarkdown(payload)` to format the message, then `setInputValue(textarea, message)`
  4. **send (D-49):** Dispatch `new KeyboardEvent('keydown', { key: 'Enter', bubbles: true, cancelable: true })` on textarea
  5. **confirm (D-50):** Call `waitForNewMessage(MESSAGE_LIST_SELECTOR, WAIT_TIMEOUT_MS)`. If false, return `{ ok: false, code: 'TIMEOUT', message: 'Message not confirmed within 5s', retriable: true }`
  6. Return `{ ok: true }`

- `function composeMarkdown(payload: { prompt: string; snapshot: Snapshot })`:
  Build prompt-first Markdown per D-39/D-40/D-41:
  ```
  const lines: string[] = [];
  if (payload.prompt) lines.push(payload.prompt, '');
  if (payload.snapshot.title) lines.push(`## ${payload.snapshot.title}`, '');
  if (payload.snapshot.url) lines.push(payload.snapshot.url, '');
  if (payload.snapshot.description) lines.push(`> ${payload.snapshot.description}`, '');
  if (payload.snapshot.create_at) lines.push(`> 采集时间: ${payload.snapshot.create_at}`, '');
  if (payload.snapshot.content) lines.push(payload.snapshot.content);
  return lines.join('\n').trim();
  ```

- `function waitForElement<T extends Element>(selector: string, timeoutMs: number): Promise<T | null>`:
  Check immediately with `document.querySelector(selector)`. If found, return. Otherwise use MutationObserver on `document.body` with `{ childList: true, subtree: true }`, checking `document.querySelector(selector)` on each mutation. Race against `setTimeout(timeoutMs)`. Always `observer.disconnect()` in both paths.

- `function waitForNewMessage(containerSelector: string, timeoutMs: number): Promise<boolean>`:
  Find container via `document.querySelector(containerSelector)`. If not found, return false. Count current children. Start MutationObserver on container with `{ childList: true }`. Resolve true when `container.children.length > initialCount`. Race against timeout. Always disconnect.

Export the `composeMarkdown` function for unit testing (named export alongside default defineContentScript).

1. Create `tests/unit/adapters/openclaw-match.spec.ts`:
   ```typescript
   import { describe, it, expect } from 'vitest';
   import { findAdapter, detectPlatformId } from '@/shared/adapters/registry';

   describe('adapters/openclaw — match (ADO-02)', () => {
     // Valid OpenClaw URLs — should ALL match
     const validUrls = [
       'http://localhost:18789/ui/chat?session=agent:main:main',
       'http://localhost:18789/chat?session=agent:main:main',
       'http://192.168.1.100:8080/ui/chat?session=agent:bot1:session1',
       'https://openclaw.example.com/ui/chat?session=agent:analyzer:daily',
       'http://localhost:18789/ui/chat?session=agent%3Amain%3Amain',
       'http://localhost:18789/ui/chat?session=agent:main:main&other=param',
       'http://localhost:18789/chat?session=agent:main:main#hash',
     ];

     for (const url of validUrls) {
       it(`matches: ${url}`, () => {
         const adapter = findAdapter(url);
         expect(adapter).not.toBeUndefined();
         expect(adapter!.id).toBe('openclaw');
       });
     }

     it('detectPlatformId returns openclaw for valid URL', () => {
       expect(detectPlatformId('http://localhost:18789/ui/chat?session=agent:main:main')).toBe('openclaw');
     });

     // Invalid URLs — should NOT match
     const invalidUrls = [
       'http://localhost:18789/ui/chat',              // no session param
       'http://localhost:18789/settings',             // wrong path
       'http://localhost:18789/',                     // root
       'https://discord.com/channels/123/456',       // discord
       'not-a-url',                                  // malformed
       '',                                           // empty
       'http://localhost:18789/ui/chat?other=value', // wrong param
     ];

     for (const url of invalidUrls) {
       it(`does NOT match: ${url || '(empty)'}`, () => {
         const adapter = findAdapter(url);
         const isOpenclaw = adapter?.id === 'openclaw';
         expect(isOpenclaw).toBe(false);
       });
     }

     it('trailing slash does NOT match (/ui/chat/ is not /ui/chat)', () => {
       // URL constructor normalizes trailing slash on paths with no file component
       // /ui/chat?session=... has query so trailing slash makes it /ui/chat/?session=...
       const adapter = findAdapter('http://localhost:18789/ui/chat/?session=agent:main:main');
       // This depends on URL constructor behavior — document either way
       // Acceptable: match or not match as long as consistent
       expect(adapter === undefined || adapter.id === 'openclaw').toBe(true);
     });
   });
   ```

2. Create `tests/unit/adapters/openclaw-compose.spec.ts`:

   ```typescript
   import { describe, it, expect } from 'vitest';
   import { composeMarkdown } from '@/entrypoints/openclaw.content';

   describe('adapters/openclaw — composeMarkdown (ADO-01, D-39, D-40, D-41)', () => {
     const fullSnapshot = {
       title: 'Test Article',
       url: 'https://example.com/article',
       description: 'A test description',
       create_at: '2026-05-01T12:00:00.000Z',
       content: '# Content\n\nParagraph here.',
     };

     it('formats prompt-first with all fields', () => {
       const result = composeMarkdown({ prompt: 'Summarize this', snapshot: fullSnapshot });
       const lines = result.split('\n');
       expect(lines[0]).toBe('Summarize this');
       expect(lines[1]).toBe('');
       expect(lines[2]).toBe('## Test Article');
       expect(result).toContain('https://example.com/article');
       expect(result).toContain('> A test description');
       expect(result).toContain('> 采集时间: 2026-05-01T12:00:00.000Z');
       expect(result).toContain('# Content\n\nParagraph here.');
     });

     it('omits empty fields entirely (D-40)', () => {
       const sparse = { title: 'Only Title', url: '', description: '', create_at: '', content: '' };
       const result = composeMarkdown({ prompt: '', snapshot: sparse });
       expect(result).toBe('## Only Title');
       expect(result).not.toContain('采集时间');
       expect(result).not.toContain('> \n');
     });

     it('omits prompt when empty', () => {
       const result = composeMarkdown({ prompt: '', snapshot: fullSnapshot });
       expect(result.startsWith('## Test Article')).toBe(true);
     });

     it('handles all fields empty except content', () => {
       const minimal = { title: '', url: '', description: '', create_at: '', content: 'Just content' };
       const result = composeMarkdown({ prompt: '', snapshot: minimal });
       expect(result).toBe('Just content');
     });

     it('no truncation applied (D-41)', () => {
       const longContent = 'x'.repeat(10_000);
       const snapshot = { ...fullSnapshot, content: longContent };
       const result = composeMarkdown({ prompt: 'p', snapshot });
       expect(result).toContain(longContent);
       expect(result.length).toBeGreaterThan(10_000);
     });
   });
   ```

|----------|-------------|
| SW -> content script (ADAPTER_DISPATCH) | SW sends dispatch payload containing user-provided prompt + snapshot content into adapter running in target page |
| content script -> target page DOM | Adapter writes text into textarea + dispatches keyboard events |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-04-02-01 | Tampering | entrypoints/openclaw.content.ts compose | accept | Content written via textarea.value (property-descriptor setter) — treated as plain text by the page. No innerHTML, no script tags. Even if prompt contains `<script>`, textarea value is not parsed as HTML. |
| T-04-02-02 | Spoofing | entrypoints/openclaw.content.ts isAdapterDispatch | mitigate | Type guard verifies message has `type === 'ADAPTER_DISPATCH'` before processing. Only SW can inject content script + send messages (chrome.scripting requires extension's own script files). |
| T-04-02-03 | Denial of Service | entrypoints/openclaw.content.ts waitForElement | mitigate | MutationObserver always disconnected via 5s timeout fallback + explicit disconnect on success path. No infinite observation possible. |
</threat_model>
