    function extractChatId(url: string): string | null {
      try {
        const u = new URL(url);
        const parts = u.pathname.split('/').filter(Boolean);
        // /a/<chatId> or /a/#<chatId>
        if (parts[0] === 'a' && parts.length >= 2) {
          return parts[parts.length - 1];
        }
        return null;
      } catch { return null; }
    }
